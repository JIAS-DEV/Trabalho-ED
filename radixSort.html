Radix Sort

O Radix sort é um algoritmo de ordenação rápido e estável que pode ser usado para ordenar itens que estão identificados por chaves únicas. Cada chave é uma cadeia de caracteres ou número, e o radix sort ordena estas chaves em qualquer ordem relacionada com a lexicografia. Na ciência da computação, radix sort é um algoritmo de ordenação que ordena inteiros processando dígitos individuais. Como os inteiros podem representar strings compostas de caracteres (como nomes ou datas) e pontos flutuantes especialmente formatados, radix sort não é limitado somente a inteiros. 

        
            package Ordenacao;

            import java.util.Arrays;

            public class AlgoritmosOrdenacao {
                long tempoInicial;
                long tempoTotal;
                boolean primeiraIteracao = true;

                public AlgoritmosOrdenacao() {
                }

                /*
                    ALGORITMOS AUXILIARES
                */

                private void troca(int[] vetor, int i, int j) {
                    int aux = vetor[i];
                    vetor[i] = vetor[j];
                    vetor[j] = aux;
                }

                /*
                    ALGORITMOS DE ORDENAÇÃO
                */
                public long radixSort(int[] vetor){ // RADIX SORT
                    tempoInicial = System.nanoTime();

                    int maximo = 0;

                    for(int i = 0; i < vetor.length; i++){
                        if(maximo < vetor[i])
                            maximo = vetor[i];
                    }

                    int digitoMax = 1;

                    while(maximo / 10 > 0){
                        digitoMax++;
                        maximo = maximo / 10;
                    }
                }
            }
